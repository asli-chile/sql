# Configuración para API del Banco Central de Chile
# Para obtener el tipo de cambio USD/CLP (dólar observado)
BANCO_CENTRAL_USER=tu_usuario_banco_central
BANCO_CENTRAL_PASS=tu_contraseña_banco_central

# Nota: Para obtener estas credenciales:
# 1. Ve a https://si3.bcentral.cl/SieteRestWS/SieteRestWS.ashx
# 2. Regístrate en el sitio del Banco Central de Chile
# 3. Solicita acceso a la API REST
# 4. Obtén tu usuario y contraseña
# 5. Agrega las variables aquí o en .env.local
#
# IMPORTANTE: Si no tienes credenciales del Banco Central, puedes usar APIs alternativas:
# - MinIndicador.cl (gratuita, sin credenciales): https://mindicador.cl/api/dolar/{dd-mm-yyyy}
# - DolarApi.com (gratuita, sin credenciales): https://cl.dolarapi.com

# Configuración para API AIS (Seguimiento de buques en tiempo real)
# aisstream.io (GRATUITO - WebSocket)
# El sistema usa WebSocket para recibir datos AIS en tiempo real
AISSTREAM_API_KEY=tu_api_key_aisstream
#
# Nota: Para obtener la API key de aisstream.io (GRATUITA):
# 1. Ve a https://aisstream.io
# 2. Regístrate y obtén tu API key gratuita
# 3. Agrega la variable AISSTREAM_API_KEY en .env.local
# 4. El servicio WebSocket se inicia automáticamente al llamar /api/vessels/start-websocket
#
# IMPORTANTE: 
# - aisstream.io es completamente GRATUITO
# - Usa WebSocket para streaming en tiempo real (wss://stream.aisstream.io/v0/stream)
# - Los datos se actualizan automáticamente en la base de datos
# - No se requieren llamadas REST costosas
